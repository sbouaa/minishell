# Manual Heredoc Syntax Error Tests
# Copy and paste these commands into your minishell

# 1. Missing delimiter tests
echo hello <<
echo hello << | cat

# 2. Empty delimiter tests  
echo hello << ""
echo hello << ''

# 3. Invalid delimiter tests
echo hello << <EOF
echo hello << >EOF
echo hello << |EOF
echo hello << <|>EOF

# 4. Consecutive heredoc tests
echo hello << << EOF
echo hello << EOF << EOF

# 5. Valid heredoc (should work)
echo hello << EOF
world
EOF

# 6. Edge cases
echo hello << 
echo hello << 	
echo hello << | cat

# 7. Mixed error cases
| << EOF
<< EOF |

# 8. Test with variables
export TEST=EOF
echo hello << $TEST
echo hello << $TEST
EOF

# 9. Test with quotes in delimiter
echo hello << "EOF"
test
EOF

echo hello << 'EOF'
test
EOF
